def tucker_phi(a, b):
    return float(np.dot(a, b) / (np.sqrt(np.dot(a, a)) * np.sqrt(np.dot(b, b))))
cong_rows = []
for c in MI_CONS:
    items = CONS[c]
    if len(items) < 3:
        continue
    try:
        f0 = FactorAnalyzer(n_factors=1, rotation=None, method='ml'); f0.fit(df.loc[df.Foreign == 0, items])
        f1 = FactorAnalyzer(n_factors=1, rotation=None, method='ml'); f1.fit(df.loc[df.Foreign == 1, items])
        l0 = np.abs(f0.loadings_[:, 0]); l1 = np.abs(f1.loadings_[:, 0])
        phi = tucker_phi(l0, l1)
        interpretation = 'Equivalent (phi >= .95)' if phi >= .95 else ('Fair (.90-.95)' if phi >= .90 else 'Non-equivalent (< .90)')
        cong_rows.append({'Construct': c, "Tucker's phi": round(phi, 3), 'Max |delta loading|': round(float(np.max(np.abs(l0 - l1))), 3), 'Interpretation': interpretation})
    except Exception:
        cong_rows.append({'Construct': c, "Tucker's phi": np.nan, 'Max |delta loading|': np.nan, 'Interpretation': 'not estimable'})
cong_df = pd.DataFrame(cong_rows)
cong_df.to_csv(os.path.join(OUT, 'MI_02_congruence.csv'), index=False, encoding='utf-8-sig')

dif_rows = []
for c in ['GLO', 'SAT_O', 'SAT_J', 'OCM']:
    items = CONS[c]
    theta = df[items].mean(axis=1)
    theta_z = (theta - theta.mean()) / theta.std()
    for item in items:
        y = (df[item] - df[item].mean()) / df[item].std()
        X = pd.DataFrame({'theta': theta_z, 'grp': df.Foreign, 'inter': theta_z * df.Foreign})
        m = sm.OLS(y, sm.add_constant(X)).fit()
        dif_rows.append({'Construct': c, 'Item': item, 'Uniform DIF (b_grp)': round(m.params['grp'], 3), 'p_uniform': round(m.pvalues['grp'], 4), 'Non-uniform (b_int)': round(m.params['inter'], 3), 'p_nonuniform': round(m.pvalues['inter'], 4), 'Flagged': 'Yes' if (m.pvalues['grp'] < .01 or m.pvalues['inter'] < .01) else 'No'})

    k_bar = float((len(df) - (k ** 2).sum() / len(df)) / (len(k) - 1))
    msb = float(g.mean().sub(df[v].mean()).pow(2).mul(k).sum() / (len(k) - 1))
    msw = float(((df[v] - df.groupby('Truong')[v].transform('mean')) ** 2).sum() / (len(df) - len(k)))
    icc1 = (msb - msw) / (msb + (k_bar - 1) * msw) if (msb + (k_bar - 1) * msw) != 0 else np.nan
    icc2 = (msb - msw) / msb if msb != 0 else np.nan
    deff = 1 + (k_bar - 1) * icc1
    icc_rows.append({'Variable': v, 'ICC(1)': round(icc1, 3), 'ICC(2)': round(icc2, 3), 'Design effect': round(deff, 2), 'Effective N': int(round(len(df) / deff)) if deff > 0 else len(df), 'Clustering matters': 'Yes' if icc1 >= .05 else 'Negligible'})
icc_df = pd.DataFrame(icc_rows)
